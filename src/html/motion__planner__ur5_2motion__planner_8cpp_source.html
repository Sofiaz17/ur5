25Matrix4d calculateTransMatrix(double th, double alpha, double distance, double a)
26{
27 Matrix4d Tij;
28 Tij << cos(th), -sin(th) * cos(alpha), sin(th) * sin(alpha), a * cos(th),
29 sin(th), cos(th) * cos(alpha), -cos(th) * sin(alpha), a * sin(th),
30 0, sin(alpha), cos(alpha), distance,
31 0, 0, 0, 1;
32 return Tij;
33}

125Vector3d lerp(double time, Vector3d p1, Vector3d p2)
126{
127 /*since interpolation evaluates which weight must be given to each vectors,
128 we need to normalize the time with reference to path_dt*/
129 const double normalizedTime = time / path_dt;
130
131 if (normalizedTime < 1)
132 // for position we need to use linear interpolation (lerp), that's why the use of quaternion is preferred
133 return (normalizedTime * p2) + ((1 - normalizedTime) * p1);
134 else
135 return p2; // zero weight is given to p1, since time exceed our delta
136}

152Quaterniond getSlerp(double time, Quaterniond q1, Quaterniond q2)
153{
154 q1.normalize();

157 if (normalizedTime < 1)
158 /*Returns the spherical linear interpolation between the two quaternions
159 *this and other at the parameter t in [0;1]*/
160 return (q1.slerp(normalizedTime, q2));

162 return q2;
